#   sudo apt-get install python3-tk

numpy            # Core numerical arrays and math
pandas           # DataFrames, CSV loading/saving
opencv-python    # Image processing (cv2)
matplotlib       # Plotting, visualization, embedded GUIs
Pillow           # Image loading/manipulation (PIL)
scipy            # Scientific functions, Hungarian algorithm, etc.
tifffile         # Efficient TIFF file support
scikit-image     # Alternative image I/O and utilities
psutil           # System memory info (used for caching optimization)
